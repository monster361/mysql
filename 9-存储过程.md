### 存储过程
存储过程是一个SQL语句集合，当主动去调用存储过程时，其中内部的SQL语句会按照逻辑执行。

### 创建存储过程
```mysql
-- 创建存储过程
delimiter //
create procedure p1()
BEGIN
    select * from t1;
END//
delimiter ;


-- 执行存储过程
call p1()

/*
对于存储过程，可以接收参数，其参数有三类：
    in        仅用于传入参数用
    out       仅用于返回值用
    inout     既可以传入又可以当作返回值
*/

-- 创建带参数的存储过程
delimiter \\
create procedure p1(
    in i1 int,
    in i2 int,
    inout i3 int,
    out r1 int
)
BEGIN
    DECLARE temp1 int;
    DECLARE temp2 int default 0;
    
    set temp1 = 1;

    set r1 = i1 + i2 + temp1 + temp2;
    
    set i3 = i3 + 100;

end\\
delimiter ;

-- 执行存储过程
set @t1 =4;  -- @t1，session级别的变量叫t1
set @t2 = 0;
CALL p1 (1, 2 ,@t1, @t2);
SELECT @t1,@t2;  -- 获取返回值
```

### 存储过程实现事物
```mysql
delimiter \\
create PROCEDURE p1(
    OUT p_return_code tinyint
)
BEGIN 
    DECLARE exit handler for sqlexception  -- 如果出现异常执行下面代码
    BEGIN 
        -- ERROR 
        set p_return_code = 1; 
        rollback;  -- 回滚
    END; 

    DECLARE exit handler for sqlwarning 
    BEGIN 
        -- WARNING 
        set p_return_code = 2; 
        rollback; 
    END;

    START TRANSACTION;  -- 开始事物
        DELETE from tb1;
        insert into tb2(name)values('seven');
    COMMIT; 

    -- SUCCESS 
    set p_return_code = 0; 

END\\
delimiter ;
```
